# aula 3 códigos
~~~
#criando

cd
mkdir (nome)
cd (nome)
git init
echo (#nome) > (nome)
git add(nome)
git commit -m "assunto"
git remote add origin (link)
git push -u origin master
git log
~~~

# aula 4 códigos
~~~
criando

git checkout -b <nome>
echo "# commit-<nome>-r" > commit-<nome>.txt
git add .
git commit -m"assunto"
git branch -r

mesclando

git checkout <nome>
git merge <nome>

deletando

git branch -d <nome>

conflitos

git pull
git add .
git commit  -m"assunto"
git log
git push origin <nome>

comandos úteis

git fetch origin <nome>
git diff <nome> origin/<nome>

~~~
